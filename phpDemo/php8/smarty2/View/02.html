{$smarty.now}
{$smarty.now|date_format:"%Y-%m-%d %H:%M:%S"}
{*时间格式化显示*} {$addr|default:'beijing'}
{*默认值*} {$baidu}
{$baidu|escape}
{*html标签转换为符合实体*} {$baidu|indent:4:'hello'}

{*内容截取*} {$talk|truncate:22:'..':true}

{*内容截取*} {$talk|truncate:22:'..':true:true|upper|lower}
